100 // This is an alternative to os_signpost_emit_with_type that allows non-compile time strings (namely
101 // for us, __FUNCTION__).
102 // The trade-off is that this doesn't allow messages to have printf-style formatting.
103 // It's fine to pass an empty string to __builtin_os_log_format_buffer_size and
104 // __builtin_os_log_format, because they return the same value for strings without any format
105 // specifiers.
106 // This is fragile, so should only be used to assist debugging and never in production.
107 #define APPLE_SIGNPOST_EMIT(log, type, spid, name, message) \
108  if (os_signpost_enabled(log)) { \
109  uint8_t _os_fmt_buf[__builtin_os_log_format_buffer_size("")]; \
110  _os_signpost_emit_with_name_impl( \
111  &__dso_handle, log, type, spid, \
112  name, message, \
113  (uint8_t *)__builtin_os_log_format(_os_fmt_buf, ""), \
114  (uint32_t)sizeof(_os_fmt_buf)); \
115  }

138  inline void traceBegin(uint32_t tag, const char* name) noexcept {
139  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
140  APPLE_SIGNPOST_EMIT(sGlobalState.systraceLog, OS_SIGNPOST_INTERVAL_BEGIN,
141  OS_SIGNPOST_ID_EXCLUSIVE, name, name)
142  }
143  }
144 
145  inline void traceEnd(uint32_t tag, const char* name) noexcept {
146  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
147  APPLE_SIGNPOST_EMIT(sGlobalState.systraceLog, OS_SIGNPOST_INTERVAL_END,
148  OS_SIGNPOST_ID_EXCLUSIVE, name, "")
149  }
150  }

164  inline void value(uint32_t tag, const char* name, int32_t value) noexcept {
165  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
166  char buf[64];
167  snprintf(buf, 64, "%s - %d", name, value);
168  APPLE_SIGNPOST_EMIT(sGlobalState.systraceLog, OS_SIGNPOST_EVENT,
169  OS_SIGNPOST_ID_EXCLUSIVE, name, buf)
170  }
171  }
172 
173  inline void value(uint32_t tag, const char* name, int64_t value) noexcept {
174  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
175  char buf[64];
176  snprintf(buf, 64, "%s - %lld", name, value);
177  APPLE_SIGNPOST_EMIT(sGlobalState.systraceLog, OS_SIGNPOST_EVENT,
178  OS_SIGNPOST_ID_EXCLUSIVE, name, buf)
179  }
180  }

182  inline void frameId(uint32_t tag, uint32_t frame) noexcept {
183  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
184  char buf[64]; \
185  snprintf(buf, 64, "frame %u", frame); \
186  APPLE_SIGNPOST_EMIT(sGlobalState.frameIdLog, OS_SIGNPOST_EVENT,
187  OS_SIGNPOST_ID_EXCLUSIVE, "frame", buf)
188  }
189  }

215 class ScopedTrace {
216  public:
217  // we don't inline this because it's relatively heavy due to a global check
218  ScopedTrace(uint32_t tag, const char* name) noexcept : mTrace(tag), mName(name), mTag(tag) {
219  mTrace.traceBegin(tag, name);
220  }
221 
222  inline ~ScopedTrace() noexcept {
223  mTrace.traceEnd(mTag, mName);
224  }
