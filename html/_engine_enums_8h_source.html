79 // The number of specialization constants that Filament reserves for its own use. These are always
80 // the first constants (from 0 to CONFIG_MAX_RESERVED_SPEC_CONSTANTS - 1).
81 // Updating this value necessitates a material version bump.
82 constexpr size_t CONFIG_MAX_RESERVED_SPEC_CONSTANTS = 16;
83 
84 // The maximum number of shadowmaps.
85 // There is currently a maximum limit of 128 shadowmaps.
86 // Factors contributing to this limit:
87 // - minspec for UBOs is 16KiB, which currently can hold a maximum of 128 entries
88 constexpr size_t CONFIG_MAX_SHADOWMAPS = 64;
89 
90 // The maximum number of shadow layers.
91 // There is currently a maximum limit of 255 layers.
92 // Several factors are contributing to this limit:
93 // - minspec for 2d texture arrays layer is 256
94 // - we're using uint8_t to store the number of layers (255 max)
95 // - nonsensical to be larger than the number of shadowmaps
96 constexpr size_t CONFIG_MAX_SHADOW_LAYERS = CONFIG_MAX_SHADOWMAPS;
97 
98 // The maximum number of shadow cascades that can be used for directional lights.
99 constexpr size_t CONFIG_MAX_SHADOW_CASCADES = 4;
100 
101 // The maximum UBO size, in bytes. This value is set to 16 KiB due to the ES3.0 spec.
102 // Note that this value constrains the maximum number of skinning bones, morph targets,
103 // instances, and shadow casting spotlights.
104 constexpr size_t CONFIG_MINSPEC_UBO_SIZE = 16384;
105 
106 // The maximum number of instances that Filament automatically creates as an optimization.
107 // Use a much smaller number for WebGL as a workaround for the following Chrome issues:
108 // https://crbug.com/1348017 Compiling GLSL is very slow with struct arrays
109 // https://crbug.com/1348363 Lighting looks wrong with D3D11 but not OpenGL
110 // Note that __EMSCRIPTEN__ is not defined when running matc, but that's okay because we're
111 // actually using a specification constant.
112 #if defined(__EMSCRIPTEN__)
113 constexpr size_t CONFIG_MAX_INSTANCES = 8;
114 #else
115 constexpr size_t CONFIG_MAX_INSTANCES = 64;
116 #endif
117 
118 // The maximum number of bones that can be associated with a single renderable.
119 // We store 32 bytes per bone. Must be a power-of-two, and must fit within CONFIG_MINSPEC_UBO_SIZE.
120 constexpr size_t CONFIG_MAX_BONE_COUNT = 256;
121 
122 // The maximum number of morph targets associated with a single renderable.
123 // Note that ES3.0 only guarantees 256 layers in an array texture.
124 // Furthermore, this is constrained by CONFIG_MINSPEC_UBO_SIZE (16 bytes per morph target).
125 constexpr size_t CONFIG_MAX_MORPH_TARGET_COUNT = 256;
126 
127 // The max number of eyes supported in stereoscopic mode.
128 // The number of eyes actually rendered is set at Engine creation time, see
129 // Engine::Config::stereoscopicEyeCount.
130 constexpr uint8_t CONFIG_MAX_STEREOSCOPIC_EYES = 4;
