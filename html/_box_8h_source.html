204  float contains(math::float3 p) const noexcept {
205  // we don't use std::max to avoid a dependency on <algorithm>

207  float d = min.x - p.x;

223  static Aabb transform(const math::mat3f& m, math::float3 const& t, const Aabb& box) noexcept {
224  // Fast AABB transformation per Jim Arvo in Graphics Gems (1990).
225  Aabb result{ t, t };
226  for (size_t col = 0; col < 3; ++col) {
227  for (size_t row = 0; row < 3; ++row) {
228  const float a = m[col][row] * box.min[col];
229  const float b = m[col][row] * box.max[col];
230  result.min[row] += a < b ? a : b;
231  result.max[row] += a < b ? b : a;
232  }
233  }
234  return result;
235  }
