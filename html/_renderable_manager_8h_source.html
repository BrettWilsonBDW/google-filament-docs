923 template<typename VECTOR, typename INDEX, typename, typename>
924 Box RenderableManager::computeAABB(
925  VECTOR const* UTILS_NONNULL vertices,
926  INDEX const* UTILS_NONNULL indices,
927  size_t count, size_t stride) noexcept {
928  math::float3 bmin(FLT_MAX);
929  math::float3 bmax(-FLT_MAX);
930  for (size_t i = 0; i < count; ++i) {
931  VECTOR const* p = reinterpret_cast<VECTOR const*>(
932  (char const*)vertices + indices[i] * stride);
933  const math::float3 v(p->x, p->y, p->z);
934  bmin = min(bmin, v);
935  bmax = max(bmax, v);
936  }
937  return Box().set(bmin, bmax);
938 }
