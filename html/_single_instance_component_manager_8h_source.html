36 /*
37  * Helper class to create single instance component managers.
38  *
39  * This handles the component's storage as a structure-of-arrays, as well
40  * as the garbage collection.
41  *
42  * This is intended to be used as base class for a real component manager. When doing so,
43  * and the real component manager is a public API, make sure to forward the public methods
44  * to the implementation.
45  *
46  */

51  // this is just to avoid using std::default_random_engine, since we're in a public header.
52  class default_random_engine {
53  uint32_t mState = 1u; // must be 0 < seed < 0x7fffffff
54  public:
55  inline uint32_t operator()() noexcept {
56  return mState = uint32_t((uint64_t(mState) * 48271u) % 0x7fffffffu);
57  }
58  };

87  // returns true if the given Entity has a component of this Manager

92  // Get instance of this Entity to be used to retrieve components
93  UTILS_NOINLINE
94  Instance getInstance(Entity e) const noexcept {
95  auto const& map = mInstanceMap;
96  // find() generates quite a bit of code
97  auto pos = map.find(e);
98  return pos != map.end() ? pos->second : 0;
99  }

101  // Returns the number of components (i.e. size of each array)
102  size_t getComponentCount() const noexcept {
103  // The array as an extra dummy component at index 0, so the visible count is 1 less.
104  return mData.size() - 1;

207  // swap only internals
208  void swap(Instance i, Instance j) noexcept {
209  assert(i);
210  assert(j);
211  if (i && j) {
212  // update the index map
213  auto& map = mInstanceMap;
214  Entity& ei = elementAt<ENTITY_INDEX>(i);
215  Entity& ej = elementAt<ENTITY_INDEX>(j);
216  std::swap(ei, ej);
217  if (ei) {
218  map[ei] = i;
219  }
220  if (ej) {
221  map[ej] = j;
222  }
223  }
224  }

226  template<typename REMOVE>
227  void gc(const EntityManager& em,
228  REMOVE&& removeComponent) noexcept {
229  gc(em, 4, std::forward<REMOVE>(removeComponent));
230  }
231 
232  template<typename REMOVE>
233  void gc(const EntityManager& em, size_t ratio,
234  REMOVE&& removeComponent) noexcept {
235  Entity const* const pEntities = begin<ENTITY_INDEX>();
236  size_t count = getComponentCount();
237  size_t aliveInARow = 0;
238  default_random_engine& rng = mRng;
239  UTILS_NOUNROLL
240  while (count && aliveInARow < ratio) {
241  assert_invariant(count == getComponentCount());
242  // note: using the modulo favorizes lower number
243  size_t const i = rng() % count;
244  Entity const entity = pEntities[i];
245  assert_invariant(entity);
246  if (UTILS_LIKELY(em.isAlive(entity))) {
247  ++aliveInARow;
248  continue;
249  }
250  removeComponent(entity);
251  aliveInARow = 0;
252  count--;
253  }
254  }

265 // Keep these outside of the class because CLion has trouble parsing them
266 template<typename ... Elements>
267 typename SingleInstanceComponentManager<Elements ...>::Instance
268 SingleInstanceComponentManager<Elements ...>::addComponent(Entity e) {
269  Instance ci = 0;
270  if (!e.isNull()) {
271  if (!hasComponent(e)) {
272  // this is like a push_back(e);
273  mData.push_back(Structure{}).template back<ENTITY_INDEX>() = e;
274  // index 0 is used when the component doesn't exist
275  ci = Instance(mData.size() - 1);
276  mInstanceMap[e] = ci;
277  } else {
278  // if the entity already has this component, just return its instance
279  ci = mInstanceMap[e];
280  }
281  }
282  assert(ci != 0);
283  return ci;
284 }

286 // Keep these outside of the class because CLion has trouble parsing them
287 template <typename ... Elements>
288 typename SingleInstanceComponentManager<Elements ...>::Instance
289 SingleInstanceComponentManager<Elements ... >::removeComponent(Entity e) {
290  auto& map = mInstanceMap;
291  auto pos = map.find(e);
292  if (UTILS_LIKELY(pos != map.end())) {
293  size_t index = pos->second;
294  assert(index != 0);
295  size_t last = mData.size() - 1;
296  if (last != index) {
297  // move the last item to where we removed this component, as to keep
298  // the array tightly packed.
299  mData.forEach([index, last](auto* p) {
300  p[index] = std::move(p[last]);
301  });
302 
303  Entity lastEntity = mData.template elementAt<ENTITY_INDEX>(index);
304  map[lastEntity] = index;
305  }
306  mData.pop_back();
307  map.erase(pos);
308  return last;
309  }
310  return 0;
311 }
