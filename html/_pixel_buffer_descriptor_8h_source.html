217  static constexpr size_t computeDataSize(PixelDataFormat format, PixelDataType type,
218  size_t stride, size_t height, size_t alignment) noexcept {
219  assert_invariant(alignment);
220 
221  if (type == PixelDataType::COMPRESSED) {
222  return 0;
223  }
224 
225  size_t n = 0;
226  switch (format) {
227  case PixelDataFormat::R:
228  case PixelDataFormat::R_INTEGER:
229  case PixelDataFormat::DEPTH_COMPONENT:
230  case PixelDataFormat::ALPHA:
231  n = 1;
232  break;
233  case PixelDataFormat::RG:
234  case PixelDataFormat::RG_INTEGER:
235  case PixelDataFormat::DEPTH_STENCIL:
236  n = 2;
237  break;
238  case PixelDataFormat::RGB:
239  case PixelDataFormat::RGB_INTEGER:
240  n = 3;
241  break;
242  case PixelDataFormat::UNUSED: // shouldn't happen (used to be rgbm)
243  case PixelDataFormat::RGBA:
244  case PixelDataFormat::RGBA_INTEGER:
245  n = 4;
246  break;
247  }
248 
249  size_t bpp = n;
250  switch (type) {
251  case PixelDataType::COMPRESSED: // Impossible -- to squash the IDE warnings
252  case PixelDataType::UBYTE:
253  case PixelDataType::BYTE:
254  // nothing to do
255  break;
256  case PixelDataType::USHORT:
257  case PixelDataType::SHORT:
258  case PixelDataType::HALF:
259  bpp *= 2;
260  break;
261  case PixelDataType::UINT:
262  case PixelDataType::INT:
263  case PixelDataType::FLOAT:
264  bpp *= 4;
265  break;
266  case PixelDataType::UINT_10F_11F_11F_REV:
267  // Special case, format must be RGB and uses 4 bytes
268  assert_invariant(format == PixelDataFormat::RGB);
269  bpp = 4;
270  break;
271  case PixelDataType::UINT_2_10_10_10_REV:
272  // Special case, format must be RGBA and uses 4 bytes
273  assert_invariant(format == PixelDataFormat::RGBA);
274  bpp = 4;
275  break;
276  case PixelDataType::USHORT_565:
277  // Special case, format must be RGB and uses 2 bytes
278  assert_invariant(format == PixelDataFormat::RGB);
279  bpp = 2;
280  break;
281  }
282 
283  size_t const bpr = bpp * stride;
284  size_t const bprAligned = (bpr + (alignment - 1)) & (~alignment + 1);
285  return bprAligned * height;
286  }
