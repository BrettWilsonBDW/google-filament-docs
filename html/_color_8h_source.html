200 // converts an RGBA color to linear space
201 // the conversion depends on the specified type
202 inline LinearColorA Color::toLinear(RgbaType type, math::float4 color) {
203  switch (type) {
204  case RgbaType::sRGB:
205  return Color::toLinear<ACCURATE>(color) * math::float4{color.a, color.a, color.a, 1.0f};
206  case RgbaType::LINEAR:
207  return color * math::float4{color.a, color.a, color.a, 1.0f};
208  case RgbaType::PREMULTIPLIED_sRGB:
209  return Color::toLinear<ACCURATE>(color);
210  case RgbaType::PREMULTIPLIED_LINEAR:
211  return color;
212  }
213 }
