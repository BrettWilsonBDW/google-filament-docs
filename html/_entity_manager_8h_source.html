51  // maximum number of entities that can exist at the same time
52  static size_t getMaxEntityCount() noexcept {
53  // because index 0 is reserved, we only have 2^GENERATION_SHIFT - 1 valid indices
54  return RAW_INDEX_COUNT - 1;
55  }
56 
57  // number of active Entities
58  size_t getEntityCount() const noexcept;

60  // Create n entities. Thread safe.
61  void create(size_t n, Entity* entities);

63  // destroys n entities. Thread safe.
64  void destroy(size_t n, Entity* entities) noexcept;
65 
66  // Create a new Entity. Thread safe.
67  // Return Entity.isNull() if the entity cannot be allocated.
68  Entity create() {
69  Entity e;
70  create(1, &e);
71  return e;
72  }
73 
74  // Destroys an Entity. Thread safe.
75  void destroy(Entity e) noexcept {
76  destroy(1, &e);
77  }
78 
79  // Return whether the given Entity has been destroyed (false) or not (true).
80  // Thread safe.
81  bool isAlive(Entity e) const noexcept {
82  assert(getIndex(e) < RAW_INDEX_COUNT);
83  return (!e.isNull()) && (getGeneration(e) == mGens[getIndex(e)]);
84  }
85 
86  // Registers a listener to be called when an entity is destroyed. Thread safe.
87  // If the listener is already registered, this method has no effect.
88  void registerListener(Listener* l) noexcept;
89 
90  // unregisters a listener.
91  void unregisterListener(Listener* l) noexcept;
92 
93 
94  /* no user serviceable parts below */
95 
96  // current generation of the given index. Use for debugging and testing.
97  uint8_t getGenerationForIndex(size_t index) const noexcept {
98  return mGens[index];
99  }

101  // singleton, can't be copied
102  EntityManager(const EntityManager& rhs) = delete;
103  EntityManager& operator=(const EntityManager& rhs) = delete;

105 #if FILAMENT_UTILS_TRACK_ENTITIES

110 private:

115  // GENERATION_SHIFT determines how many simultaneous Entities are available, the
116  // minimum memory requirement is 2^GENERATION_SHIFT bytes.
117  static constexpr const int GENERATION_SHIFT = 17;
118  static constexpr const size_t RAW_INDEX_COUNT = (1 << GENERATION_SHIFT);
119  static constexpr const Entity::Type INDEX_MASK = (1 << GENERATION_SHIFT) - 1u;
120 
121  static inline Entity::Type getGeneration(Entity e) noexcept {
122  return e.getId() >> GENERATION_SHIFT;
123  }
124  static inline Entity::Type getIndex(Entity e) noexcept {
125  return e.getId() & INDEX_MASK;
126  }
127  static inline Entity::Type makeIdentity(Entity::Type g, Entity::Type i) noexcept {
128  return (g << GENERATION_SHIFT) | (i & INDEX_MASK);
129  }
130 
131  // stores the generation of each index.
132  uint8_t * const mGens;
133 };
