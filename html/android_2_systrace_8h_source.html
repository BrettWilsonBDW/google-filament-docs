119  inline void traceBegin(uint32_t tag, const char* name) noexcept {
120  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
121  beginSection(this, name);
122  }
123  }
124 
125  inline void traceEnd(uint32_t tag) noexcept {
126  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
127  endSection(this);
128  }
129  }
130 
131  inline void asyncBegin(uint32_t tag, const char* name, int32_t cookie) noexcept {
132  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
133  beginAsyncSection(this, name, cookie);
134  }
135  }
136 
137  inline void asyncEnd(uint32_t tag, const char* name, int32_t cookie) noexcept {
138  if (tag && UTILS_UNLIKELY(mIsTracingEnabled)) {
139  endAsyncSection(this, name, cookie);
140  }
141  }
142 
143  inline void value(uint32_t tag, const char* name, int32_t value) noexcept {

223 class ScopedTrace {
224 public:

226  ScopedTrace(uint32_t tag, const char* name) noexcept: mTrace(tag), mTag(tag) {
227  mTrace.traceBegin(tag, name);
228  }
229 
230  inline ~ScopedTrace() noexcept {

232  }
