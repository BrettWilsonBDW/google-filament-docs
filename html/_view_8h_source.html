769  template<typename T, void(T::*method)(PickingQueryResult const&)>
770  void pick(uint32_t x, uint32_t y, T* UTILS_NONNULL instance,
771  backend::CallbackHandler* UTILS_NULLABLE handler = nullptr) noexcept {
772  PickingQuery& query = pick(x, y, [](PickingQueryResult const& result, PickingQuery* pq) {
773  (static_cast<T*>(pq->storage[0])->*method)(result);
774  }, handler);
775  query.storage[0] = instance;
776  }
777 
789  template<typename T, void(T::*method)(PickingQueryResult const&)>
790  void pick(uint32_t x, uint32_t y, T instance,
791  backend::CallbackHandler* UTILS_NULLABLE handler = nullptr) noexcept {
792  static_assert(sizeof(instance) <= sizeof(PickingQuery::storage), "user data too large");
793  PickingQuery& query = pick(x, y, [](PickingQueryResult const& result, PickingQuery* pq) {
794  T* const that = static_cast<T*>(reinterpret_cast<void*>(pq->storage));
795  (that->*method)(result);
796  that->~T();
797  }, handler);
798  new(query.storage) T(std::move(instance));
799  }

810  template<typename T>
811  void pick(uint32_t x, uint32_t y, T functor,
812  backend::CallbackHandler* UTILS_NULLABLE handler = nullptr) noexcept {
813  static_assert(sizeof(functor) <= sizeof(PickingQuery::storage), "functor too large");
814  PickingQuery& query = pick(x, y, handler,
815  (PickingQueryResultCallback)[](PickingQueryResult const& result, PickingQuery* pq) {
816  T* const that = static_cast<T*>(reinterpret_cast<void*>(pq->storage));
817  that->operator()(result);
818  that->~T();
819  });
820  new(query.storage) T(std::move(functor));
821  }
