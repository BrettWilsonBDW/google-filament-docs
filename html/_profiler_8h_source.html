38 class Profiler {
39 public:
40  enum {
41  INSTRUCTIONS = 0, // must be zero
42  CPU_CYCLES = 1,
43  DCACHE_REFS = 2,
44  DCACHE_MISSES = 3,
45  BRANCHES = 4,
46  BRANCH_MISSES = 5,
47  ICACHE_REFS = 6,
48  ICACHE_MISSES = 7,
49 
50  // Must be last one
51  EVENT_COUNT
52  };
53 
54  enum {
55  EV_CPU_CYCLES = 1u << CPU_CYCLES,
56  EV_L1D_REFS = 1u << DCACHE_REFS,
57  EV_L1D_MISSES = 1u << DCACHE_MISSES,
58  EV_BPU_REFS = 1u << BRANCHES,
59  EV_BPU_MISSES = 1u << BRANCH_MISSES,
60  EV_L1I_REFS = 1u << ICACHE_REFS,
61  EV_L1I_MISSES = 1u << ICACHE_MISSES,
62  // helpers
63  EV_L1D_RATES = EV_L1D_REFS | EV_L1D_MISSES,
64  EV_L1I_RATES = EV_L1I_REFS | EV_L1I_MISSES,
65  EV_BPU_RATES = EV_BPU_REFS | EV_BPU_MISSES,
66  };

68  Profiler() noexcept; // must call resetEvents()
69  explicit Profiler(uint32_t eventMask) noexcept;
70  ~Profiler() noexcept;

73  Profiler(Profiler&& rhs) = delete;

83  bool isValid() const { return mCountersFd[0] >= 0; }

85  class Counters {
86  friend class Profiler;
87 
88  uint64_t nr;
89  uint64_t time_enabled;
90  uint64_t time_running;
91  struct {
92  uint64_t value;
93  uint64_t id;
94  } counters[Profiler::EVENT_COUNT];
95 
96  friend Counters operator-(Counters lhs, const Counters& rhs) noexcept {
97  lhs.nr -= rhs.nr;
98  lhs.time_enabled -= rhs.time_enabled;
99  lhs.time_running -= rhs.time_running;
100  for (size_t i = 0; i < EVENT_COUNT; ++i) {
101  lhs.counters[i].value -= rhs.counters[i].value;
102  }
103  return lhs;
104  }
105 
106  public:
107  uint64_t getInstructions() const { return counters[INSTRUCTIONS].value; }
108  uint64_t getCpuCycles() const { return counters[CPU_CYCLES].value; }
109  uint64_t getL1DReferences() const { return counters[DCACHE_REFS].value; }
110  uint64_t getL1DMisses() const { return counters[DCACHE_MISSES].value; }
111  uint64_t getL1IReferences() const { return counters[ICACHE_REFS].value; }
112  uint64_t getL1IMisses() const { return counters[ICACHE_MISSES].value; }
113  uint64_t getBranchInstructions() const { return counters[BRANCHES].value; }
114  uint64_t getBranchMisses() const { return counters[BRANCH_MISSES].value; }
115 
116  std::chrono::duration<uint64_t, std::nano> getWallTime() const {
117  return std::chrono::duration<uint64_t, std::nano>(time_enabled);
118  }
119 
120  std::chrono::duration<uint64_t, std::nano> getRunningTime() const {
121  return std::chrono::duration<uint64_t, std::nano>(time_running);
122  }
123 
124  double getIPC() const noexcept {
125  uint64_t cpuCycles = getCpuCycles();
126  uint64_t instructions = getInstructions();
127  return double(instructions) / double(cpuCycles);
128  }
129 
130  double getCPI() const noexcept {
131  uint64_t cpuCycles = getCpuCycles();
132  uint64_t instructions = getInstructions();
133  return double(cpuCycles) / double(instructions);
134  }
135 
136  double getL1DMissRate() const noexcept {
137  uint64_t cacheReferences = getL1DReferences();
138  uint64_t cacheMisses = getL1DMisses();
139  return double(cacheMisses) / double(cacheReferences);
140  }
141 
142  double getL1DHitRate() const noexcept {
143  return 1.0 - getL1DMissRate();
144  }
145 
146  double getL1IMissRate() const noexcept {
147  uint64_t cacheReferences = getL1IReferences();
148  uint64_t cacheMisses = getL1IMisses();
149  return double(cacheMisses) / double(cacheReferences);
150  }
151 
152  double getL1IHitRate() const noexcept {
153  return 1.0 - getL1IMissRate();
154  }
155 
156  double getBranchMissRate() const noexcept {
157  uint64_t branchReferences = getBranchInstructions();
158  uint64_t branchMisses = getBranchMisses();
159  return double(branchMisses) / double(branchReferences);
160  }
161 
162  double getBranchHitRate() const noexcept {
163  return 1.0 - getBranchMissRate();
164  }
165 
166  double getMPKI(uint64_t misses) const noexcept {
167  return (misses * 1000.0) / getInstructions();
168  }
169  };

197 #endif // __linux__

199  bool hasBranchRates() const noexcept {
200  return (mCountersFd[BRANCHES] >= 0) && (mCountersFd[BRANCH_MISSES] >= 0);
201  }
202 
203  bool hasICacheRates() const noexcept {
204  return (mCountersFd[ICACHE_REFS] >= 0) && (mCountersFd[ICACHE_MISSES] >= 0);
205  }
