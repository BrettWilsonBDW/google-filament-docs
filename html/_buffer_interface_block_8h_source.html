63  struct FieldInfo {
64  utils::CString name; // name of this field
65  uint16_t offset; // offset in "uint32_t" of this field in the buffer
66  uint8_t stride; // stride in "uint32_t" to the next element
67  Type type; // type of this field
68  bool isArray; // true if the field is an array
69  uint32_t size; // size of the array in elements, or 0 if not an array
70  Precision precision; // precision of this field
71  backend::FeatureLevel minFeatureLevel; // below this feature level, this field is not needed
72  utils::CString structName; // name of this field structure if type is STRUCT
73  utils::CString sizeName; // name of the size parameter in the shader
74  // returns offset in bytes of this field (at index if an array)
75  inline size_t getBufferOffset(size_t index = 0) const {
76  assert_invariant(index < std::max(1u, size));
77  return (offset + stride * index) * sizeof(uint32_t);
78  }
79  };
