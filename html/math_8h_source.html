31 UTILS_PUBLIC T cubicSpline(const T& vert0, const T& tang0, const T& vert1, const T& tang1, float t) {
32  float tt = t * t, ttt = tt * t;
33  float s2 = -2 * ttt + 3 * tt, s3 = ttt - tt;
34  float s0 = 1 - s2, s1 = s3 - tt + t;
35  T p0 = vert0;
36  T m0 = tang0;
37  T p1 = vert1;
38  T m1 = tang1;
39  return s0 * p0 + s1 * m0 * t + s2 * p1 + s3 * m1 * t;
40 }

42 UTILS_PUBLIC inline void decomposeMatrix(const filament::math::mat4f& mat, filament::math::float3* translation,
43  filament::math::quatf* rotation, filament::math::float3* scale) {
44  using namespace filament::math;
45 
46  // Extract translation.
47  *translation = mat[3].xyz;
48 
49  // Extract upper-left for determinant computation.
50  const float a = mat[0][0];
51  const float b = mat[0][1];
52  const float c = mat[0][2];
53  const float d = mat[1][0];
54  const float e = mat[1][1];
55  const float f = mat[1][2];
56  const float g = mat[2][0];
57  const float h = mat[2][1];
58  const float i = mat[2][2];
59  const float A = e * i - f * h;
60  const float B = f * g - d * i;
61  const float C = d * h - e * g;
62 
63  // Extract scale.
64  const float det(a * A + b * B + c * C);
65  float scalex = length(float3({a, b, c}));
66  float scaley = length(float3({d, e, f}));
67  float scalez = length(float3({g, h, i}));
68  float3 s = { scalex, scaley, scalez };
69  if (det < 0) {
70  s = -s;
71  }
72  *scale = s;
73 
74  // Remove scale from the matrix if it is not close to zero.
75  mat4f clone = mat;
76  if (std::abs(det) > std::numeric_limits<float>::epsilon()) {
77  clone[0] /= s.x;
78  clone[1] /= s.y;
79  clone[2] /= s.z;
80  // Extract rotation
81  *rotation = clone.toQuaternion();
82  } else {
83  // Set to identity if close to zero
84  *rotation = quatf(1.0f);
85  }
86 }

88 UTILS_PUBLIC inline filament::math::mat4f composeMatrix(const filament::math::float3& translation,
89  const filament::math::quatf& rotation, const filament::math::float3& scale) {
90  float tx = translation[0];
91  float ty = translation[1];
92  float tz = translation[2];
93  float qx = rotation[0];
94  float qy = rotation[1];
95  float qz = rotation[2];
96  float qw = rotation[3];
97  float sx = scale[0];
98  float sy = scale[1];
99  float sz = scale[2];
100  return filament::math::mat4f(
101  (1 - 2 * qy*qy - 2 * qz*qz) * sx,
102  (2 * qx*qy + 2 * qz*qw) * sx,
103  (2 * qx*qz - 2 * qy*qw) * sx,
104  0.f,
105  (2 * qx*qy - 2 * qz*qw) * sy,
106  (1 - 2 * qx*qx - 2 * qz*qz) * sy,
107  (2 * qy*qz + 2 * qx*qw) * sy,
108  0.f,
109  (2 * qx*qz + 2 * qy*qw) * sz,
110  (2 * qy*qz - 2 * qx*qw) * sz,
111  (1 - 2 * qx*qx - 2 * qy*qy) * sz,
112  0.f, tx, ty, tz, 1.f);
113 }

115 inline filament::math::mat3f matrixFromUvTransform(const float offset[2], float rotation,
116  const float scale[2]) {
117  float tx = offset[0];
118  float ty = offset[1];
119  float sx = scale[0];
120  float sy = scale[1];
121  float c = cos(rotation);
122  float s = sin(rotation);
123  return filament::math::mat3f(sx * c, sx * s, tx, -sy * s, sy * c, ty, 0.0f, 0.0f, 1.0f);
124 };
