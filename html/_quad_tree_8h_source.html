39 static inline constexpr uint16_t morton(uint8_t x, uint8_t y) noexcept {
40  uint32_t r = x | (uint32_t(y) << 16);
41  r = (r | (r << 4u)) & 0x0f0f0f0fu;
42  r = (r | (r << 2u)) & 0x33333333u;
43  r = (r | (r << 1u)) & 0x55555555u;
44  return uint16_t(r | (r >> 15u));
45 }

95  // Simple fixed capacity stack
96  template<typename TYPE, size_t CAPACITY,
97  typename = typename std::enable_if<std::is_trivial<TYPE>::value>::type>
98  class stack {
99  TYPE mElements[CAPACITY];
100  size_t mSize = 0;
101  public:
102  bool empty() const noexcept { return mSize == 0; }
103  void push(TYPE const& v) noexcept {
104  assert_invariant(mSize < CAPACITY);
105  mElements[mSize++] = v;
106  }
107  void pop() noexcept {
108  assert_invariant(mSize > 0);
109  --mSize;
110  }
111  const TYPE& back() const noexcept {
112  return mElements[mSize - 1];
113  }
114  };
115 
116 public:
117  using code_t = uint8_t;
118 
119  struct NodeId {
120  int8_t l; // height of the node or -1 if invalid
121  code_t code; // morton code of the node
122  };
123 
124  enum class TraversalResult {
125  EXIT, // end traversal
126  RECURSE, // proceed with the children
127  SKIP_CHILDREN // skip children
128  };
129 
130  static inline constexpr size_t height() noexcept {
131  return HEIGHT;
132  }

143  template<typename Process,
144  typename = typename std::enable_if<
145  std::is_invocable_r_v<TraversalResult, Process, NodeId>>::type>
146  static void traverse(int8_t l, code_t code, size_t maxHeight, Process&& process) noexcept {
147  assert_invariant(maxHeight < height());
148  const int8_t h = int8_t(maxHeight);
149  stack<NodeId, 4 * height()> stack;
150  stack.push({ l, code });
151  while (!stack.empty()) {
152  NodeId curr = stack.back();
153  stack.pop();
154  TraversalResult r = process(curr);
155  if (r == TraversalResult::EXIT) {
156  break;
157  }
158  if (r == TraversalResult::RECURSE && curr.l < h) {
159  for (size_t c = 0; c < 4; c++) {
160  stack.push({
161  int8_t(curr.l + 1u),
162  code_t((curr.code << 2u) | (3u - c))
163  });
164  }
165  }
166  }
167  }
168 
169  template<typename Node,
170  typename = typename std::enable_if<
171  std::is_invocable_r_v<TraversalResult, Node, NodeId>>::type>
172  static void traverse(int8_t l, code_t code, Node&& process) noexcept {
173  traverse(l, code, height() - 1, std::forward<Node>(process));
174  }
175 };
