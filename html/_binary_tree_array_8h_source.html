30 class BinaryTreeArray {
31 
32  // Simple fixed capacity stack
33  template<typename TYPE, size_t CAPACITY,
34  typename = typename std::enable_if<std::is_trivial<TYPE>::value>::type>
35  class stack {
36  TYPE mElements[CAPACITY];
37  size_t mSize = 0;
38  public:
39  bool empty() const noexcept { return mSize == 0; }
40  void push(TYPE const& v) noexcept {
41  assert(mSize < CAPACITY);
42  mElements[mSize++] = v;
43  }
44  void pop() noexcept {
45  assert(mSize > 0);
46  --mSize;
47  }
48  const TYPE& back() const noexcept {
49  return mElements[mSize - 1];
50  }
51  };
52 
53 public:
54  static size_t count(size_t height) noexcept { return (1u << height) - 1; }
55  static size_t left(size_t i, size_t /*height*/) noexcept { return i + 1; }
56  static size_t right(size_t i, size_t height) noexcept {
57  return i + (size_t(1) << (height - 1));
58  }
59 
60  // this builds the depth-first binary tree array top down (post-order)
61  template<typename Leaf, typename Node>
62  static void traverse(size_t height, Leaf leaf, Node node) noexcept {
63 
64  struct TNode {
65  uint32_t index;
66  uint32_t col;
67  uint32_t height;
68  uint32_t next;
69 
70  bool isLeaf() const noexcept { return height == 1; }
71  size_t left() const noexcept { return BinaryTreeArray::left(index, height); }
72  size_t right() const noexcept { return BinaryTreeArray::right(index, height); }
73  };
74 
75  stack<TNode, 16> stack;
76  stack.push(TNode{ 0, 0, (uint32_t)height, (uint32_t)count(height) });
77 
78  uint32_t prevLeft = 0;
79  uint32_t prevRight = 0;
80  uint32_t prevIndex = 0;
81  while (!stack.empty()) {
82  TNode const* const UTILS_RESTRICT curr = &stack.back();
83  const bool isLeaf = curr->isLeaf();
84  const uint32_t index = curr->index;
85  const uint32_t l = (uint32_t)curr->left();
86  const uint32_t r = (uint32_t)curr->right();
87 
88  if (prevLeft == index || prevRight == index) {
89  if (!isLeaf) {
90  // the 'next' node of our left node's right descendants is our right child
91  stack.push({ l, 2 * curr->col, curr->height - 1, r });
92  }
93  } else if (l == prevIndex) {
94  if (!isLeaf) {
95  // the 'next' node of our right child is our own 'next' sibling
96  stack.push({ r, 2 * curr->col + 1, curr->height - 1, curr->next });
97  }
98  } else {
99  if (!isLeaf) {
100  node(index, l, r, curr->next);
101  } else {
102  leaf(index, curr->col, curr->next);
103  }
104  stack.pop();
105  }
106 
107  prevLeft = l;
108  prevRight = r;
109  prevIndex = index;
110  }
111  }
112 };
