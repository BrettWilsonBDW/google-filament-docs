86  /*
87  * Generally we generate GLSL that will be converted to SPIRV, optimized and then
88  * transpiled to the backend's language such as MSL, ESSL300, GLSL410 or SPIRV, in this
89  * case the generated GLSL uses ESSL310 or GLSL450 and has Vulkan semantics and
90  * TargetLanguage::SPIRV must be used.
91  *
92  * However, in some cases (e.g. when no optimization is asked) we generate the *final* GLSL
93  * directly, this GLSL must be ESSL300 or GLSL410 and cannot use any Vulkan syntax, for this
94  * situation we use TargetLanguage::GLSL. In this case TargetApi is guaranteed to be OPENGL.
95  *
96  * Note that TargetLanguage::GLSL is not the common case, as it is generally not used in
97  * release builds.
98  *
99  * Also note that glslang performs semantics analysis on whichever GLSL ends up being generated.
100  */
101  enum class TargetLanguage {
102  GLSL, // GLSL with OpenGL 4.1 / OpenGL ES 3.0 semantics
103  SPIRV // GLSL with Vulkan semantics
104  };

155 // Utility function that looks at an Engine backend to determine TargetApi
156 inline constexpr MaterialBuilderBase::TargetApi targetApiFromBackend(
157  filament::backend::Backend backend) noexcept {
158  using filament::backend::Backend;
159  using TargetApi = MaterialBuilderBase::TargetApi;
160  switch (backend) {
161  case Backend::DEFAULT: return TargetApi::ALL;
162  case Backend::OPENGL: return TargetApi::OPENGL;
163  case Backend::VULKAN: return TargetApi::VULKAN;
164  case Backend::METAL: return TargetApi::METAL;
165  case Backend::NOOP: return TargetApi::OPENGL;
166  }
