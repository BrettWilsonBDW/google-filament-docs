65  // Size is chosen so that we can store at least std::function<>
66  // the alignas() qualifier ensures we're multiple of a cache-line.
67  static constexpr size_t JOB_STORAGE_SIZE_BYTES =
68  sizeof(std::function<void()>) > 48 ? sizeof(std::function<void()>) : 48;
69  static constexpr size_t JOB_STORAGE_SIZE_WORDS =
70  (JOB_STORAGE_SIZE_BYTES + sizeof(void*) - 1) / sizeof(void*);

72  // keep it first, so it's correctly aligned with all architectures
73  // this is where we store the job's data, typically a std::function<>
74  // v7 | v8

76  JobFunc function; // 4 | 8
77  uint16_t parent; // 2 | 2
78  std::atomic<uint16_t> runningJobCount = { 1 }; // 2 | 2
79  mutable std::atomic<uint16_t> refCount = { 1 }; // 2 | 2

112  /*
113  * Job creation examples:
114  * ----------------------
115  *
116  * struct Functor {
117  * uintptr_t storage[6];
118  * void operator()(JobSystem&, Jobsystem::Job*);
119  * } functor;
120  *
121  * struct Foo {
122  * uintptr_t storage[6];
123  * void method(JobSystem&, Jobsystem::Job*);
124  * } foo;
125  *
126  * Functor and Foo size muse be <= uintptr_t[6]
127  *
128  * createJob()
129  * createJob(parent)
130  * createJob<Foo, &Foo::method>(parent, &foo)
131  * createJob<Foo, &Foo::method>(parent, foo)
132  * createJob<Foo, &Foo::method>(parent, std::ref(foo))
133  * createJob(parent, functor)
134  * createJob(parent, std::ref(functor))
135  * createJob(parent, [ up-to 6 uintptr_t ](JobSystem*, Jobsystem::Job*){ })
136  *
137  * Utility functions:
138  * ------------------
139  * These are less efficient, but handle any size objects using the heap if needed.
140  * (internally uses std::function<>), and don't require the callee to take
141  * a (JobSystem&, Jobsystem::Job*) as parameter.
142  *
143  * struct BigFoo {
144  * uintptr_t large[16];
145  * void operator()();
146  * void method(int answerToEverything);
147  * static void exec(BigFoo&) { }
148  * } bigFoo;
149  *
150  * jobs::createJob(js, parent, [ any-capture ](int answerToEverything){}, 42);
151  * jobs::createJob(js, parent, &BigFoo::method, &bigFoo, 42);
152  * jobs::createJob(js, parent, &BigFoo::exec, std::ref(bigFoo));
153  * jobs::createJob(js, parent, bigFoo);
154  * jobs::createJob(js, parent, std::ref(bigFoo));
155  * etc...
156  *
157  * struct SmallFunctor {
158  * uintptr_t storage[3];
159  * void operator()(T* data, size_t count);
160  * } smallFunctor;
161  *
162  * jobs::parallel_for(js, data, count, [ up-to 3 uintptr_t ](T* data, size_t count) { });
163  * jobs::parallel_for(js, data, count, smallFunctor);
164  * jobs::parallel_for(js, data, count, std::ref(smallFunctor));
165  *
166  */

173  // creates a job from a KNOWN method pointer w/ object passed by pointer
174  // the caller must ensure the object will outlive the Job
175  template<typename T, void(T::*method)(JobSystem&, Job*)>
176  Job* createJob(Job* parent, T* data) noexcept {
177  Job* job = create(parent, +[](void* storage, JobSystem& js, Job* job) {
178  T* const that = static_cast<T*>(reinterpret_cast<void**>(storage)[0]);
179  (that->*method)(js, job);
180  });
181  if (job) {
182  job->storage[0] = data;
183  }
184  return job;
185  }
186 
187  // creates a job from a KNOWN method pointer w/ object passed by value
188  template<typename T, void(T::*method)(JobSystem&, Job*)>
189  Job* createJob(Job* parent, T data) noexcept {
190  static_assert(sizeof(data) <= sizeof(Job::storage), "user data too large");
191  Job* job = create(parent, [](void* storage, JobSystem& js, Job* job) {
192  T* const that = static_cast<T*>(storage);
193  (that->*method)(js, job);
194  that->~T();
195  });
196  if (job) {
197  new(job->storage) T(std::move(data));
198  }
199  return job;
200  }

202  // creates a job from a functor passed by value
203  template<typename T>
204  Job* createJob(Job* parent, T functor) noexcept {
205  static_assert(sizeof(functor) <= sizeof(Job::storage), "functor too large");
206  Job* job = create(parent, [](void* storage, JobSystem& js, Job* job){
207  T* const that = static_cast<T*>(storage);
208  that->operator()(js, job);
209  that->~T();
210  });
211  if (job) {
212  new(job->storage) T(std::move(functor));
213  }
214  return job;

406 // IMPORTANT: these are less efficient to call and may perform heap allocation

409 template<typename CALLABLE, typename ... ARGS>
410 JobSystem::Job* createJob(JobSystem& js, JobSystem::Job* parent,
411  CALLABLE&& func, ARGS&&... args) noexcept {
412  struct Data {
413  std::function<void()> f;
414  // Renaming the method below could cause an Arrested Development.
415  void gob(JobSystem&, JobSystem::Job*) noexcept { f(); }
416  } user{ std::bind(std::forward<CALLABLE>(func),
417  std::forward<ARGS>(args)...) };
418  return js.createJob<Data, &Data::gob>(parent, std::move(user));
419 }

440 template<typename S, typename F>
441 struct ParallelForJobData {
442  using SplitterType = S;
443  using Functor = F;
444  using JobData = ParallelForJobData;
445  using size_type = uint32_t;
446 
447  ParallelForJobData(size_type start, size_type count, uint8_t splits,
448  Functor functor,
449  const SplitterType& splitter) noexcept
450  : start(start), count(count),
451  functor(std::move(functor)),
452  splits(splits),
453  splitter(splitter) {
454  }
455 
456  void parallelWithJobs(JobSystem& js, JobSystem::Job* parent) noexcept {
457  assert(parent);
458 
459  // this branch is often miss-predicted (it both sides happen 50% of the calls)
460 right_side:
461  if (splitter.split(splits, count)) {
462  const size_type lc = count / 2;
463  JobData ld(start, lc, splits + uint8_t(1), functor, splitter);
464  JobSystem::Job* l = js.createJob<JobData, &JobData::parallelWithJobs>(parent, std::move(ld));
465  if (UTILS_UNLIKELY(l == nullptr)) {
466  // couldn't create a job, just pretend we're done splitting
467  goto execute;
468  }
469 
470  // start the left side before attempting the right side, so we parallelize in case
471  // of job creation failure -- rare, but still.
472  js.run(l);
473 
474  // don't spawn a job for the right side, just reuse us -- spawning jobs is more
475  // costly than we'd like.
476  start += lc;
477  count -= lc;
478  ++splits;
479  goto right_side;
480 
481  } else {
482 execute:
483  // we're done splitting, do the real work here!
484  functor(start, count);
485  }
486  }
487 
488 private:
489  size_type start; // 4
490  size_type count; // 4
491  Functor functor; // ?
492  uint8_t splits; // 1
493  SplitterType splitter; // 1
494 };
