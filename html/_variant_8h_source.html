34 // IMPORTANT: update filterVariant() when adding more variants
35 // Also be sure to update formatVariantString inside CommonWriter.cpp
36 struct Variant {
37  using type_t = uint8_t;
38 
39  Variant() noexcept = default;
40  Variant(Variant const& rhs) noexcept = default;
41  Variant& operator=(Variant const& rhs) noexcept = default;
42  constexpr explicit Variant(type_t key) noexcept : key(key) { }
43 
44 
45  // DIR: Directional Lighting
46  // DYN: Dynamic Lighting
47  // SRE: Shadow Receiver
48  // SKN: Skinning
49  // DEP: Depth only
50  // FOG: Fog
51  // PCK: Picking (depth variant only)
52  // VSM: Variance shadow maps
53  // STE: Instanced stereo rendering
54  //
55  // X: either 1 or 0
56  // +-----+-----+-----+-----+-----+-----+-----+-----+
57  // Variant | STE | VSM | FOG | DEP | SKN | SRE | DYN | DIR | 256
58  // +-----+-----+-----+-----+-----+-----+-----+-----+
59  // PCK
60  //
61  // Standard variants:
62  // +-----+-----+-----+-----+-----+-----+-----+-----+
63  // | STE | VSM | FOG | 0 | SKN | SRE | DYN | DIR | 128 - 44 = 84
64  // +-----+-----+-----+-----+-----+-----+-----+-----+
65  // Vertex shader X 0 0 0 X X X X
66  // Fragment shader 0 X X 0 0 X X X
67  // Fragment SSR 0 1 0 0 0 1 0 0
68  // Reserved X 1 1 0 X 1 0 0 [ -4]
69  // Reserved X 0 X 0 X 1 0 0 [ -8]
70  // Reserved X 1 X 0 X 0 X X [-32]
71  //
72  // Depth variants:
73  // +-----+-----+-----+-----+-----+-----+-----+-----+
74  // | STE | VSM | PCK | 1 | SKN | 0 | 0 | 0 | 16 - 4 = 12
75  // +-----+-----+-----+-----+-----+-----+-----+-----+
76  // Vertex depth X X 0 1 X 0 0 0
77  // Fragment depth 0 X X 1 0 0 0 0
78  // Reserved X 1 1 1 X 0 0 0 [ -4]
79  //
80  // 96 variants used, 160 reserved (256 - 96)
81  //
82  // note: a valid variant can be neither a valid vertex nor a valid fragment variant
83  // (e.g.: FOG|SKN variants), the proper bits are filtered appropriately,
84  // see filterVariantVertex(), filterVariantFragment().
85 
86  type_t key = 0u;
87 
88  // when adding more bits, update FRenderer::CommandKey::draw::materialVariant as needed
89  // when adding more bits, update VARIANT_COUNT
90  static constexpr type_t DIR = 0x01; // directional light present, per frame/world position
91  static constexpr type_t DYN = 0x02; // point, spot or area present, per frame/world position
92  static constexpr type_t SRE = 0x04; // receives shadows, per renderable
93  static constexpr type_t SKN = 0x08; // GPU skinning and/or morphing
94  static constexpr type_t DEP = 0x10; // depth only variants
95  static constexpr type_t FOG = 0x20; // fog (standard)
96  static constexpr type_t PCK = 0x20; // picking (depth)
97  static constexpr type_t VSM = 0x40; // variance shadow maps
98  static constexpr type_t STE = 0x80; // instanced stereo
99 
100  // special variants (variants that use the reserved space)
101  static constexpr type_t SPECIAL_SSR = VSM | SRE; // screen-space reflections variant
102 
103  static constexpr type_t STANDARD_MASK = DEP;
104  static constexpr type_t STANDARD_VARIANT = 0u;
105 
106  // the depth variant deactivates all variants that make no sense when writing the depth
107  // only -- essentially, all fragment-only variants.
108  static constexpr type_t DEPTH_MASK = DEP | SRE | DYN | DIR;
109  static constexpr type_t DEPTH_VARIANT = DEP;
110 
111  // this mask filters out the lighting variants
112  static constexpr type_t UNLIT_MASK = STE | SKN | FOG;
113 
114  // returns raw variant bits
115  inline bool hasDirectionalLighting() const noexcept { return key & DIR; }
116  inline bool hasDynamicLighting() const noexcept { return key & DYN; }
117  inline bool hasSkinningOrMorphing() const noexcept { return key & SKN; }
118  inline bool hasStereo() const noexcept { return key & STE; }
119 
120  inline void setDirectionalLighting(bool v) noexcept { set(v, DIR); }
121  inline void setDynamicLighting(bool v) noexcept { set(v, DYN); }
122  inline void setShadowReceiver(bool v) noexcept { set(v, SRE); }
123  inline void setSkinning(bool v) noexcept { set(v, SKN); }
124  inline void setFog(bool v) noexcept { set(v, FOG); }
125  inline void setPicking(bool v) noexcept { set(v, PCK); }
126  inline void setVsm(bool v) noexcept { set(v, VSM); }
127  inline void setStereo(bool v) noexcept { set(v, STE); }
128 
129  inline static constexpr bool isValidDepthVariant(Variant variant) noexcept {
130  // Can't have VSM and PICKING together with DEPTH variants
131  constexpr type_t RESERVED_MASK = VSM | PCK | DEP | SRE | DYN | DIR;
132  constexpr type_t RESERVED_VALUE = VSM | PCK | DEP;
133  return ((variant.key & DEPTH_MASK) == DEPTH_VARIANT) &&
134  ((variant.key & RESERVED_MASK) != RESERVED_VALUE);
135  }
136 
137  inline static constexpr bool isValidStandardVariant(Variant variant) noexcept {
138  // can't have shadow receiver if we don't have any lighting
139  constexpr type_t RESERVED0_MASK = VSM | FOG | SRE | DYN | DIR;
140  constexpr type_t RESERVED0_VALUE = VSM | FOG | SRE;
141 
142  // can't have shadow receiver if we don't have any lighting
143  constexpr type_t RESERVED1_MASK = VSM | SRE | DYN | DIR;
144  constexpr type_t RESERVED1_VALUE = SRE;
145 
146  // can't have VSM without shadow receiver
147  constexpr type_t RESERVED2_MASK = VSM | SRE;
148  constexpr type_t RESERVED2_VALUE = VSM;
149 
150  return ((variant.key & STANDARD_MASK) == STANDARD_VARIANT) &&
151  ((variant.key & RESERVED0_MASK) != RESERVED0_VALUE) &&
152  ((variant.key & RESERVED1_MASK) != RESERVED1_VALUE) &&
153  ((variant.key & RESERVED2_MASK) != RESERVED2_VALUE);
154  }
155 
156  inline static constexpr bool isVertexVariant(Variant variant) noexcept {
157  return filterVariantVertex(variant) == variant;
158  }
159 
160  inline static constexpr bool isFragmentVariant(Variant variant) noexcept {
161  return filterVariantFragment(variant) == variant;
162  }
163 
164  static constexpr bool isReserved(Variant variant) noexcept {
165  return !isValid(variant);
166  }
167 
168  static constexpr bool isValid(Variant variant) noexcept {
169  return isValidStandardVariant(variant) || isValidDepthVariant(variant);
170  }
171 
172  inline static constexpr bool isSSRVariant(Variant variant) noexcept {
173  return (variant.key & (STE | VSM | DEP | SRE | DYN | DIR)) == (VSM | SRE);
174  }
175 
176  inline static constexpr bool isVSMVariant(Variant variant) noexcept {
177  return !isSSRVariant(variant) && ((variant.key & VSM) == VSM);
178  }
179 
180  inline static constexpr bool isShadowReceiverVariant(Variant variant) noexcept {
181  return !isSSRVariant(variant) && ((variant.key & SRE) == SRE);
182  }
183 
184  inline static constexpr bool isFogVariant(Variant variant) noexcept {
185  return (variant.key & (FOG | DEP)) == FOG;
186  }
187 
188  inline static constexpr bool isPickingVariant(Variant variant) noexcept {
189  return (variant.key & (PCK | DEP)) == (PCK | DEP);
190  }
191 
192  inline static constexpr bool isStereoVariant(Variant variant) noexcept {
193  return (variant.key & STE) == STE;
194  }
195 
196  static constexpr Variant filterVariantVertex(Variant variant) noexcept {
197  // filter out vertex variants that are not needed. For e.g. fog doesn't affect the
198  // vertex shader.
199  if ((variant.key & STANDARD_MASK) == STANDARD_VARIANT) {
200  if (isSSRVariant(variant)) {
201  variant.key &= ~(VSM | SRE);
202  }
203  return variant & (STE | SKN | SRE | DYN | DIR);
204  }
205  if ((variant.key & DEPTH_MASK) == DEPTH_VARIANT) {
206  // Only VSM, skinning, and stereo affect the vertex shader's DEPTH variant
207  return variant & (STE | VSM | SKN | DEP);
208  }
209  return {};
210  }
211 
212  static constexpr Variant filterVariantFragment(Variant variant) noexcept {
213  // filter out fragment variants that are not needed. For e.g. skinning doesn't
214  // affect the fragment shader.
215  if ((variant.key & STANDARD_MASK) == STANDARD_VARIANT) {
216  return variant & (VSM | FOG | SRE | DYN | DIR);
217  }
218  if ((variant.key & DEPTH_MASK) == DEPTH_VARIANT) {
219  // Only VSM & PICKING affects the fragment shader's DEPTH variant
220  return variant & (VSM | PCK | DEP);
221  }
222  return {};
223  }
224 
225  static constexpr Variant filterVariant(Variant variant, bool isLit) noexcept {
226  // special case for depth variant
227  if (isValidDepthVariant(variant)) {
228  if (!isLit) {
229  // if we're unlit, we never need the VSM variant
230  return variant & ~VSM;
231  }
232  return variant;
233  }
234  if (isSSRVariant(variant)) {
235  return variant;
236  }
237  if (!isLit) {
238  // when the shading mode is unlit, remove all the lighting variants
239  return variant & UNLIT_MASK;
240  }
241  // if shadow receiver is disabled, turn off VSM
242  if (!(variant.key & SRE)) {
243  return variant & ~VSM;
244  }
245  return variant;
246  }
247 
248  constexpr bool operator==(Variant rhs) const noexcept {
249  return key == rhs.key;
250  }
251 
252  constexpr bool operator!=(Variant rhs) const noexcept {
253  return key != rhs.key;
254  }
255 
256  constexpr Variant operator & (type_t rhs) const noexcept {
257  return Variant(key & rhs);
258  }
259 
260  static Variant filterUserVariant(
261  Variant variant, UserVariantFilterMask filterMask) noexcept;
262 
263 private:
264  void set(bool v, type_t mask) noexcept {
265  key = (key & ~mask) | (v ? mask : type_t(0));
266  }
267 };
